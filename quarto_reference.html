r_fibonacci <- function(n) {
    if (n <= 1) {
        return(n)
    } else {
        return(r_fibonacci(n - 1) + r_fibonacci(n - 2))
    }
}

Running Python

Python code
import os,sys,time
def py_fibonacci(n):
  if n <= 1:
    return n
  else:
    return(py_fibonacci(n-1) + py_fibonacci(n-2))
py_fibonacci(10)
55
